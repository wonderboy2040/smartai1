findLastBearCandle() =>
    idx = -1
    for i = 1 to 20
        if close[i] < open[i]
            idx := i
            break
    idx

    idx = -1
    for i = 1 to 20
        if close[i] > open[i]
            idx := i
            break
    idx

if (bullBOS or bullCHoCH) and showOB
    i = findLastBearCandle()
    if i >= 0
        obTop = math.max(open[i], close[i])
        obBot = low[i]
        b = box.new(left=bar_index - i, top=obTop, right=bar_index + 20, bottom=obBot, bgcolor=bullOBCol, border_color=color.new(bullCol, 70), extend=extend.right)
        array.push(bullOBs, b)
        array.push(bullOBtop, obTop)
        array.push(bullOBbot, obBot)
        if array.size(bullOBs) > maxOB
            box.delete(array.shift(bullOBs))
            array.shift(bullOBtop)
            array.shift(bullOBbot)

if (bearBOS or bearCHoCH) and showOB
    i = findLastBullCandle()
    if i >= 0
        obTop = high[i]
        obBot = math.min(open[i], close[i])
        b = box.new(left=bar_index - i, top=obTop, right=bar_index + 20, bottom=obBot, bgcolor=bearOBCol, border_color=color.new(bearCol, 70), extend=extend.right)

nearBullZone() =>
    found = false
    if array.size(bullOBtop) > 0
        for i = 0 to array.size(bullOBtop) - 1
            t = array.get(bullOBtop, i)
            b = array.get(bullOBbot, i)
            if low <= t and high >= b
                found := true
                break
    if not found and array.size(bullFVGtop) > 0
        for i = 0 to array.size(bullFVGtop) - 1
            t = array.get(bullFVGtop, i)
            b = array.get(bullFVGbot, i)
            if low <= t and high >= b
                found := true
                break
    found

nearBearZone() =>
    found = false
    if array.size(bearOBtop) > 0
        for i = 0 to array.size(bearOBtop) - 1
            t = array.get(bearOBtop, i)
            b = array.get(bearOBbot, i)
            if high >= b and low <= t
                found := true
                break
    if not found and array.size(bearFVGtop) > 0
        for i = 0 to array.size(bearFVGtop) - 1
            t = array.get(bearFVGtop, i)
            b = array.get(bearFVGbot, i)
            if high >= b and low <= t
                found := true
                break
    found

structBear = bearBOS or bearCHoCH

    sellSignal := structBear and recentBearSweep and nearBearZone()
else if sensitivity == "Balanced"
    buySignal  := structBull and (recentBullSweep or nearBullZone())
    sellSignal := structBear and (recentBearSweep or nearBearZone())
else
    buySignal  := structBull
    sellSignal := structBear

f_ascending(arr) =>
    ok = array.size(arr) >= 3
    if ok
        for i = 1 to array.size(arr) - 1
            if array.get(arr, i) <= array.get(arr, i - 1)
                ok := false
                break
    ok

f_descending(arr) =>
    ok = array.size(arr) >= 3
    if ok
        for i = 1 to array.size(arr) - 1
            if array.get(arr, i) >= array.get(arr, i - 1)
                ok := false
                break
    ok

// ============================================================================
// === ALGOALPHA TREND FILTER (Kalman + Custom Supertrend) ====================
// ============================================================================
// Ported from AlgoAlpha "Range Filtered Trend Signals" (v6 → v5).
// Detects ranging vs trending markets and provides Kalman bands for SL.
//
//   trend       =  1 (price > upper band)   |  -1 (price < lower band)
//   ktrend      =  1 (Supertrend bullish)   |  -1 (Supertrend bearish)
//   trendConfirmed = ktrend * trend ==  1   (both agree → TRENDING)
//   isRanging      = ktrend * trend == -1   (disagree    → RANGING, no trade)
// ============================================================================

// Kalman filter — adaptive smoothing of close price
f_kalman(a, b, alpha, beta) =>
    var float v1 = na
    var float v2 = 1.0
    var float v3 = alpha * b
    var float v4 = 0.0
    var float v5 = na

    if na(v1)
        v1 := a[1]

    v5 := v1
    v4 := v2 / (v2 + v3)
    v1 := v5 + v4 * (a - v5)
    v2 := (1 - v4) * v2 + beta / b
    v1

// Custom Supertrend on the Kalman line
f_pineSupertrend(k, factor, atrPeriod) =>
    src = k
    atr = ta.atr(atrPeriod)
    upperBand = src + factor * atr
    lowerBand = src - factor * atr
    prevLowerBand = nz(lowerBand[1])
    prevUpperBand = nz(upperBand[1])

    lowerBand := lowerBand > prevLowerBand or k[1] < prevLowerBand ? lowerBand : prevLowerBand
    upperBand := upperBand < prevUpperBand or k[1] > prevUpperBand ? upperBand : prevUpperBand
    int _direction = na
    float superTrend = na
    prevSuperTrend = superTrend[1]
    if na(atr[1])
        _direction := 1
    else if prevSuperTrend == prevUpperBand
        _direction := k > upperBand ? -1 : 1
    else
        _direction := k < lowerBand ? 1 : -1
    superTrend := _direction == -1 ? lowerBand : upperBand
    [superTrend, _direction]

// Compute Kalman line + Supertrend direction
k = f_kalman(close, kalmanPeriod, kalmanAlpha, kalmanBeta)
[supertrend, stDirection] = f_pineSupertrend(k, stFactor, stAtrPeriod)

// Volatility-adjusted bands — used for SL anchoring and range visualization
vola         = ta.wma(high - low, 200)
kalmanUpper  = k + vola * kalmanDev
kalmanLower  = k - vola * kalmanDev

// Short-term trend state (price vs Kalman bands)

if close > kalmanUpper
    trendAA := 1
else if close < kalmanLower
    trendAA := -1

// Long-term trend state (inverted from raw Supertrend direction per AlgoAlpha)
ktrend = 0
if stDirection < 0
    ktrend := 1
else if stDirection > 0
    ktrend := -1

// Master filter flags
trendConfirmed = ktrend * trendAA == 1     // both agree → TRENDING
isRanging      = ktrend * trendAA == -1    // disagree  → RANGING

// String label for dashboard
